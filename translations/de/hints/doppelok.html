Okay, okay! Das ist die richtige Einstellung!=) Option zur Lösung der Aufgabe mit einer for Schleife und yield Ausdruck. Eine solche Funktion wird auch als Generator betrachtet. 

from collections.abc import Iterable


def chunking_by(items: list[int], size: int) -> Iterable[list[int]]:

    for i in range(0, len(items), size):
        yield items[i:i + size]
